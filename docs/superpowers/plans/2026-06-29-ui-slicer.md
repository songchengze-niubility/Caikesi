# ui-slicer Implementation Plan

> 收尾状态：已实现。本文保留为实现过程计划；最终能力和边界以 `docs/superpowers/specs/2026-06-29-ui-slicer-design.md` 与 `tools/ui-slicer/index.html` 为准。实际版本在原矩形裁切基础上增加了轻量边缘去底、吸管、每框遮罩画笔和可选写入文件夹，但仍保持单文件、零依赖、无服务、无 AI。

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 一个自包含的本地 HTML 页,手动框选把 UI 图裁成矩形 PNG,通过下载/复制交给 Figma。

**Architecture:** 单个 `tools/ui-slicer/index.html`(CSS/JS 内联,零依赖、零构建、零服务)。内部按职责分函数块:图片加载、viewport 坐标变换、box store(框模型 + localStorage)、canvas 渲染、导出。所有框坐标存源图整数像素;canvas 用 `setTransform` 之外的手动 image↔screen 变换绘制。

**Tech Stack:** 原生 HTML / CSS / JavaScript、Canvas 2D、`canvas.toBlob`、`navigator.clipboard` + `ClipboardItem`、`localStorage`。无任何第三方库。

## Global Constraints

- **零依赖 / 零构建 / 零服务**:只能用浏览器原生 API,不引入 npm 包、不需要打包、不起本地服务进程。
- **单文件**:全部代码在 `tools/ui-slicer/index.html` 内,CSS/JS 内联。
- **非目标(绝不实现)**:AI/VLM/SAM/rembg、语义级自动拆图、alpha 质检、审查包、ZIP 打包、layout.json、工程文件存盘、组件库/Prefab、Cocos 衔接生成、本地服务。实际实现允许浏览器 Canvas 内的轻量边缘去底和人工遮罩修边。
- **产出**:只产出 PNG 像素(下载文件 + 剪贴板),不写任何坐标/元数据文件。矩形裁切**连背景一起裁**。
- **验证方式**:本工具是交互式 canvas 应用,spec 已定为**纯手动浏览器验证**(无遗留复杂纯逻辑,不建测试框架)。因此每个任务以"在浏览器里操作 + 观察预期"代替 TDD 的失败测试循环。
- **提交策略**:本项目 `CLAUDE.md` 规定**只在用户明确要求时提交/推送**。每个任务做到"可提交状态",提交动作等用户批准;提交时用中文 message,结尾带 `Co-Authored-By: Claude Opus 4.8 <noreply@anthropic.com>`。
- **坐标变换**(全程统一,勿改名):`imageToScreen(ix,iy)→{x,y}`、`screenToImage(sx,sy)→{x,y}`,关系 `screen = image*scale + offset`。
- **框模型字段**(全程统一):`{ id, name, x, y, w, h }`,x/y/w/h 为源图整数像素。

const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');
const hint = document.getElementById('hint');
const zoomLabel = document.getElementById('zoomLabel');

const state = {
  img:null, imgName:'image', imgW:0, imgH:0,
  scale:1, offsetX:0, offsetY:0,
  boxes:[], selectedId:null, nextIndex:1,
};

function imageToScreen(ix,iy){ return { x: ix*state.scale + state.offsetX, y: iy*state.scale + state.offsetY }; }
function screenToImage(sx,sy){ return { x: (sx-state.offsetX)/state.scale, y: (sy-state.offsetY)/state.scale }; }

function resizeCanvas(){
  const r = canvas.parentElement.getBoundingClientRect();
  canvas.width = Math.max(1, Math.round(r.width));
  canvas.height = Math.max(1, Math.round(r.height));
  render();
}

function fitView(){
  const pad = 24;
  const sx = (canvas.width - pad*2) / state.imgW;
  const sy = (canvas.height - pad*2) / state.imgH;
  state.scale = Math.min(sx, sy);
  state.offsetX = (canvas.width - state.imgW*state.scale)/2;
  state.offsetY = (canvas.height - state.imgH*state.scale)/2;
}

  ctx.clearRect(0,0,canvas.width,canvas.height);

}

  b.x = Math.max(0, Math.min(b.x, state.imgW));
  b.y = Math.max(0, Math.min(b.y, state.imgH));
  b.w = Math.max(0, Math.min(b.w, state.imgW - b.x));
  b.h = Math.max(0, Math.min(b.h, state.imgH - b.y));

}

```javascript
const HSIZE = 7; // 手柄屏幕像素
function handlePoints(b){
  const a = imageToScreen(b.x, b.y);
  const w = b.w*state.scale, h = b.h*state.scale;
  return {
    nw:{x:a.x,     y:a.y},     n:{x:a.x+w/2, y:a.y},     ne:{x:a.x+w, y:a.y},
    w_:{x:a.x,     y:a.y+h/2},                            e:{x:a.x+w, y:a.y+h/2},
    sw:{x:a.x,     y:a.y+h},   s:{x:a.x+w/2, y:a.y+h},   se:{x:a.x+w, y:a.y+h},
  };
}
function drawHandles(b){
  const pts = handlePoints(b);
  ctx.fillStyle = '#fff'; ctx.strokeStyle = '#1e90ff'; ctx.lineWidth = 1;
  for(const k in pts){
    ctx.fillRect(pts[k].x-HSIZE/2, pts[k].y-HSIZE/2, HSIZE, HSIZE);
    ctx.strokeRect(pts[k].x-HSIZE/2, pts[k].y-HSIZE/2, HSIZE, HSIZE);
  }
}
function hitHandle(b, sx, sy){
  const pts = handlePoints(b);
  for(const k in pts){ if(Math.abs(sx-pts[k].x)<=HSIZE && Math.abs(sy-pts[k].y)<=HSIZE) return k; }
  return null;
}
function hitBox(sx, sy){
  const p = screenToImage(sx, sy);
  // 后画的在上层,倒序命中
  for(let i=state.boxes.length-1;i>=0;i--){
    const b = state.boxes[i];
    if(p.x>=b.x && p.x<=b.x+b.w && p.y>=b.y && p.y<=b.y+b.h) return b;
  }
  return null;
}
function selectBox(id){ state.selectedId = id; render(); syncSelPanel && syncSelPanel(); }
function deleteBox(id){
  state.boxes = state.boxes.filter(b=>b.id!==id);
  if(state.selectedId===id) state.selectedId = null;
  render(); syncSelPanel && syncSelPanel();
}
```

function cropCanvas(b){
  const x=Math.max(0,Math.min(b.x,state.imgW)), y=Math.max(0,Math.min(b.y,state.imgH));
  const w=Math.max(1,Math.min(b.w,state.imgW-x)), h=Math.max(1,Math.min(b.h,state.imgH-y));
  const c=document.createElement('canvas'); c.width=Math.round(w); c.height=Math.round(h);
  c.getContext('2d').drawImage(state.img, x,y,w,h, 0,0,c.width,c.height);
  return c;
}
function blobOf(b){ return new Promise(res => cropCanvas(b).toBlob(res,'image/png')); }

async function downloadBox(b, fname){
  const blob = await blobOf(b);
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = (fname || safeName(b.name)) + '.png';
  document.body.appendChild(a); a.click(); a.remove();
  setTimeout(()=>URL.revokeObjectURL(a.href), 1500);
}
const canCopy = !!(navigator.clipboard && window.ClipboardItem);
async function copyBox(b){
  if(!canCopy){ alert('复制图片需 Chrome 或 Edge'); return; }
  await navigator.clipboard.write([ new ClipboardItem({ 'image/png': await blobOf(b) }) ]);
}
async function downloadAll(){
  const used = new Map();
  for(const b of state.boxes){
    let n = safeName(b.name); const c=(used.get(n)||0)+1; used.set(n,c);
    await downloadBox(b, c>1 ? `${n}_${c}` : n);
  }
}
function updateExportEnabled(){
  const has = !!state.img && state.boxes.length>0;
  document.getElementById('downloadAllBtn').disabled = !has;
  const sel = !!getBox(state.selectedId);
  document.getElementById('dlBtn').disabled = !sel;
  document.getElementById('copyBtn').disabled = !sel || !canCopy;
}

document.getElementById('dlBtn').addEventListener('click', ()=>{ const b=getBox(state.selectedId); if(b) downloadBox(b); });
document.getElementById('copyBtn').addEventListener('click', ()=>{ const b=getBox(state.selectedId); if(b) copyBox(b); });
document.getElementById('downloadAllBtn').addEventListener('click', downloadAll);
if(!canCopy) document.getElementById('copyBtn').title = '复制图片需 Chrome 或 Edge';
